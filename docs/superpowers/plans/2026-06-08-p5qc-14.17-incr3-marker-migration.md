# 14.17 Increment 3 — host-local marker→overlay capture migration

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax.

**Goal:** On the first install after a host adopts the markerless model, capture each
live host-local marker body into a `local.yaml` overlay span — before increment 2's
deploy blanket-strips (and would otherwise delete) it — so the file converges
markerless in one install, idempotently and revertibly.

**Architecture:** M2 (capture-only). A new on-install migration reads the live
host-local marker bodies, writes them as `at-end-of-file` OVERLAY spans into
`local.yaml` (atomic ruamel write), then the SAME install re-resolves the host-local
overlay (idempotent anchor-keyed fold) so increment 2's deploy renders markerless.
The migration never touches the live file — deploy strips it, and the transition
captures `file_pre[live]`=markered → `file_post[live]`=markerless. Approved spec:
`~/.claude/plans/cached-forging-kernighan.md`.

**Tech Stack:** Python 3.12, Pydantic v2, ruamel.yaml; existing `setforge.sections`,
`setforge.overlay_migration`, `setforge.overlay_inject`, `setforge.migrations._yaml_ops`,
`setforge.source`, `setforge.config`, `setforge.cli._install_helpers` / `install`.

---

## In plain terms

Today host-local sections live as `<!-- ... host-local NAME -->` marker pairs in the
deployed file, with per-host text inside (preserved across installs by section-merge,
never stored in `local.yaml`). Increment 2 made deploy *delete* every host-local
marker pair. So on the first install, before deploy runs, we must copy each body into
`local.yaml` as an overlay span — otherwise the per-host text is lost. After that one
install the file is markerless, the body lives in `local.yaml`, and re-installing is a
no-op.

"""Read host-local user-section marker bodies from a deployed live file (14.17).

Increment-3 capture migration: the per-host body lives only in the live file's
host-local marker region. This module reads those bodies so the install can move
them into local.yaml overlay spans BEFORE deploy's blanket strip deletes them.
"""

from __future__ import annotations

from setforge.sections import (
    SectionSemantics,
    _BodyLine,
    _EndMarker,
    _StartMarker,
    _walk_markers,
)


def extract_host_local_marker_bodies(text: str) -> dict[str, str]:
    """Return ``{name: body}`` for every HOST_LOCAL marker region in ``text``.

    Body is the exact bytes between the markers (trailing newline included, up
    to but not including the end-marker line), matching ``extract_sections``.
    Shared regions are ignored. Raises ``ValueError`` on a duplicate host-local
    name (silently dropping one would lose a per-host body).
    """
    bodies: dict[str, str] = {}
    current: str | None = None
    lines: list[str] = []
    for event in _walk_markers(text, allow_legacy=True):
        match event:
            case _StartMarker(semantics=SectionSemantics.HOST_LOCAL, name=name):
                current = name
                lines = []
            case _EndMarker(semantics=SectionSemantics.HOST_LOCAL, key=key):
                if key in bodies:
                    raise ValueError(f"duplicate host-local section name {key!r}")
                bodies[key] = "".join(lines)
                current = None
            case _BodyLine(line=line) if current is not None:
                lines.append(line)
            case _:
                pass
    return bodies
```

from pathlib import Path

from ruamel.yaml.comments import CommentedMap, CommentedSeq

from setforge.migrations._yaml_ops import atomic_write_yaml, yaml_rt
from setforge.overlay_inject import canonical_body


def build_overlay_span_node(name: str, body: str) -> CommentedMap:
    """Build an at-end-of-file host-local OVERLAY span YAML node (canonical body)."""
    anchor = CommentedMap()
    anchor["kind"] = "at-end-of-file"
    payload = CommentedMap()
    payload["anchor"] = anchor
    payload["body"] = canonical_body(body)
    entry = CommentedMap()
    entry["anchor"] = name
    entry["kind"] = "overlay"
    entry["semantics"] = "host-local"
    entry["overlay"] = payload
    return entry


def _existing_overlay_anchors(tracked_file: CommentedMap) -> set[str]:
    spans = tracked_file.get("spans")
    if not isinstance(spans, CommentedSeq):
        return set()
    return {
        str(s.get("anchor"))
        for s in spans
        if isinstance(s, CommentedMap) and s.get("kind") == "overlay"
    }


def append_overlay_spans(
    local_path: Path, additions: dict[str, list[tuple[str, str]]]
) -> int:
    """Append at-EOF host-local OVERLAY spans to local.yaml; return count written.

    Presence-check skips any ``(file_id, name)`` already present as an overlay
    span anchor (crash-resume / idempotency). No-op (no write) when nothing new.
    Atomic ruamel round-trip preserves comments, order, and file mode.
    """
    if not additions or not local_path.exists():
        return 0
    yaml = yaml_rt()
    with local_path.open("r", encoding="utf-8") as fh:
        data = yaml.load(fh)
    if not isinstance(data, CommentedMap):
        return 0
    tracked_files = data.get("tracked_files")
    if not isinstance(tracked_files, CommentedMap):
        return 0
    written = 0
    for file_id, entries in additions.items():
        tf = tracked_files.get(file_id)
        if not isinstance(tf, CommentedMap):
            # local.yaml has no overlay block for this id yet: create one.
            tf = CommentedMap()
            tracked_files[file_id] = tf
        existing = _existing_overlay_anchors(tf)
        spans = tf.get("spans")
        if not isinstance(spans, CommentedSeq):
            spans = CommentedSeq()
            tf["spans"] = spans
        for name, body in entries:
            if name in existing:
                continue
            spans.append(build_overlay_span_node(name, body))
            written += 1
    if written == 0:
        return 0
    atomic_write_yaml(local_path, data)
    return written
```

```python
@dataclass(slots=True, frozen=True)
class HostLocalMarkerMigration:
    local_path: Path
    pre_text: str | None
    migrated: bool
    names_by_file: dict[str, list[str]]


def migrate_host_local_markers_on_install(
    cfg: Config,
    resolved: ResolvedProfile,
    repo_root: Path,
    *,
    local_config_path: Path | None = None,
) -> HostLocalMarkerMigration:
    """Capture live host-local marker bodies → at-EOF OVERLAY spans in local.yaml.

    Disk + record only: writes local.yaml (atomic) and returns the pre-migration
    text for the transition seed. Does NOT mutate cfg or strip live (deploy does
    both). Empty bodies are dropped (deploy strips the empty markers). Idempotent
    via append_overlay_spans' presence-check.
    """
    from setforge import host_local_marker_migration as hlm
    from setforge import source

    local_path = (
        local_config_path if local_config_path is not None else source.LOCAL_CONFIG_PATH
    )
    additions: dict[str, list[tuple[str, str]]] = {}
    names_by_file: dict[str, list[str]] = {}
    for tf_id in resolved.tracked_files:
        tracked_file = cfg.tracked_files[tf_id]
        if not tracked_file.preserve_user_sections:
            continue
        src = resolve_src(tracked_file, repo_root)
        if src.suffix.lower() not in {".md", ".markdown"}:
            continue
        dst = resolve_dst(tracked_file)
        if not dst.exists():
            continue
        bodies = hlm.extract_host_local_marker_bodies(dst.read_text(encoding="utf-8"))
        populated = [(n, b) for n, b in bodies.items() if b.strip()]
        if populated:
            additions[tf_id] = populated
            names_by_file[tf_id] = [n for n, _ in populated]
    if not additions:
        return HostLocalMarkerMigration(local_path, None, False, {})
    pre_text: str | None
    try:
        pre_text = local_path.read_text(encoding="utf-8")
    except FileNotFoundError:
        pre_text = None
    written = hlm.append_overlay_spans(local_path, additions)
    if written == 0:
        # All names already present (crash-resume) — nothing new on disk.
        return HostLocalMarkerMigration(local_path, pre_text, False, {})
    typer.secho(
        f"{local_path}: captured host-local section(s) "
        f"{sorted(n for ns in names_by_file.values() for n in ns)} into markerless "
        "local.yaml overlay spans. To restore the pre-migration state, run "
        "`setforge revert`.",
        err=True,
        fg=typer.colors.YELLOW,
    )
    return HostLocalMarkerMigration(local_path, pre_text, True, names_by_file)


def seed_host_local_marker_snapshot(
    migration: HostLocalMarkerMigration,
    overlay_migration: OverlaySpanMigration,
    dst_paths: list[Path],
    file_pre: dict[Path, str | None],
) -> None:
    """Record the local.yaml capture in the transition (revert lockstep).

    No-op when nothing was captured. Appends local.yaml to dst_paths and seeds
    file_pre with the genuine pre-migration text — EARLIEST-WINS: if the 10.2
    seed already set file_pre for this path (overlay_migration.migrated), keep
    that value (it is pre-everything); otherwise use this migration's pre_text.
    """
    if not migration.migrated:
        return
    if migration.local_path not in dst_paths:
        dst_paths.append(migration.local_path)
    if not overlay_migration.migrated or migration.local_path not in file_pre:
        file_pre[migration.local_path] = migration.pre_text
```
